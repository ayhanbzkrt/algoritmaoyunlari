Sayı Tahmin Oyunu

1 ile 100 arasında bir sayı tahmin edin ve yapay zekanın doğru tahmini yapmasını sağlayın.

import random

def sayi_tahmin_oyunu():
    gizli_sayi = random.randint(1, 100)
    tahmin = None

    print("1 ile 100 arasında bir sayı tahmin et!")

    while tahmin != gizli_sayi:
        tahmin = int(input("Tahminin nedir? "))
       
        if tahmin < gizli_sayi:
            print("Daha büyük bir sayı tahmin et.")
        elif tahmin > gizli_sayi:
            print("Daha küçük bir sayı tahmin et.")
        else:
            print("Tebrikler, doğru tahmin ettin!")

sayi_tahmin_oyunu()

Bubble Sort Görselleştirme Oyunu

Bu oyunda, bir dizi sayının bubble sort algoritması ile sıralanmasını izleyin. Bubble sort, en basit sıralama algoritmalarından biridir ve her adımda komşu elemanları karşılaştırarak sıralama yapar.

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]):
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

arr = [64, 34, 25, 12, 22, 11, 90]
sorted_arr = bubble_sort(arr)
print(f"Sıralı dizi: {sorted_arr}")

from textblob import TextBlob

def metin_siniflandirma(metin):
    analiz = TextBlob(metin)
    if analiz.sentiment.polarity > 0:
        return "Pozitif"
    elif analiz.sentiment.polarity < 0:
        return "Negatif"
    else:
        return "Nötr"

metin = "Bu oyun harika!"
sonuc = metin_siniflandirma(metin)

Mantık Kapıları Simülasyonu

Bu oyunda, temel mantık kapılarını (AND, OR, NOT) kullanarak bir mantık devresi tasarlayın ve sonuçlarını izleyin. Mantık kapıları, dijital devrelerin temel bileşenleridir ve iki veya daha fazla giriş sinyalini belirli kurallara göre işler.

def and_gate(a, b):
    return a & b

def or_gate(a, b):
    return a | b

def not_gate(a):
    return ~a & 1

a = 1
b = 0
and_result = and_gate(a, b)
or_result = or_gate(a, b)
not_result = not_gate(a)
print(f"AND: ${and_result}, OR: ${or_result}, NOT: ${not_result}")

from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score

def veri_kumesi_siniflandirma():
    iris = load_iris()
    X_train, X_test, y_train, y_test = train_test_split(iris.data, iris.target, test_size=0.2)
    knn = KNeighborsClassifier(n_neighbors=3)
    knn.fit(X_train, y_train)
    y_pred = knn.predict(X_test)
    dogruluk = accuracy_score(y_test, y_pred)
    print(f"Sınıflandırma doğruluğu: {dogruluk}")

veri_kumesi_siniflandirma()
